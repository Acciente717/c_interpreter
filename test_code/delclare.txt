void sayhello()
{
    int h = 104; int e = 101; int l = 108; int o = 111;
    int space = 32; int w = 119; int r = 114; int d = 100;
    int excl = 33; int cr = 10;
    cputchar(h); cputchar(e); cputchar(l); cputchar(l); cputchar(o);
    cputchar(space); cputchar(w); cputchar(o); cputchar(r); cputchar(l);
    cputchar(d); cputchar(excl); cputchar(cr);
}

int wrapper(int times)
{
    while (times > 0)
    {
        sayhello();
        times = times - 1;
    }
    return times;
}

int main()
{
    cputint(wrapper(5 + 5));
    cputchar(10);
    return 0;
}
